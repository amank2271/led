#include "stdio.h"
#include "string.h"

#define IN 1
#define OUT 0
#define MAXLINE 1000

int getlineu(char s[], int lim);
void copy(char to[], char from[]);
void reverse(char s[]);
int main()
{
  int len;
  int max;
  char line[MAXLINE];
  char longest[MAXLINE];
  max = 0;
  while ((len = getlineu(line, MAXLINE)) > 0)
  {
    if (len > max)
    {
      max = len;
      copy(longest, line);
    }
    if (max > 0)
    {
      printf("%s\n", longest);
    }
    reverse(longest);
  }

  return 0;
}

int getlineu(char s[], int lim)
{
  int c, i;
  for (i = 0; i < lim - 1 && (c = getchar()) != EOF && c != '\n'; i++)
  {
    s[i] = (char)c;
  }
  if (c == '\n')
  {
    s[i] = (char)c;
    ++i;
  }
  s[i] = '\0';

  return (i);
}

void copy(char to[], char from[])
{
  int i = 0;
  while ((to[i] = from[i]) != '\0')
  {
    ++i;
  }
}

void reverse(char s[])
{
  int c, i, j;

  for (i = 0, j = strlen(s) - 1; i < j; i++, j--)
  {
    c = s[i];
    s[i] = s[j];
    s[j] = (char)c;
  }
  printf("%s\n", s);
}
